"""Constants used in the public API."""

from enum import Enum


MAJOR_VERSION = "0"
MINOR_VERSION = "4"
PATCH_VERSION = "1a1"
__short_version__ = "{}.{}".format(MAJOR_VERSION, MINOR_VERSION)
__version__ = "{}.{}".format(__short_version__, PATCH_VERSION)

    Unknown = 0

class DeviceModel(Enum):
    """Hardware device models."""

    Unknown = 0
    Gen2 = 1
    Gen3 = 2
    Gen4 = 3
    Gen4K = 4


class FeatureState(Enum):
    """State of a particular feature."""

    # Feature is supported by device but it is not known if it is available or not
    Unknown = 0

    # Device does not support this feature
    Unsupported = 1

    # Feature is supported by device but not available now (e.g. "pause" is
    # unavailable if nothing is playing)
    Unavailable = 2

    # Feature is supported and available
    Available = 3
